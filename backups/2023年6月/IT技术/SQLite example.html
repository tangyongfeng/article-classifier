using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Data.SQLite;

namespace ConsoleApplication1
{
    class Program
    {
        static void Main( string[] args)
        {
            string constr= "Data Source=mydb.db;Version=3";
            SQLiteConnection conn = new SQLiteConnection (constr);
            SQLiteCommand Comm = new SQLiteCommand ();
            conn.Open();
            Console.WriteLine( "open sucess!\n");
            Comm.Connection = conn;
            Comm.CommandText = "drop table TESTTABLE;CREATE TABLE [TESTTABLE](id int,name varchar(255))";
            Comm.ExecuteNonQuery();
            for ( int i = 2; i < 1000; i++)
            {
                Comm.CommandText= string.Format( "insert into TESTTABLE values ({0},'真的啊');" , i);
                Comm.ExecuteNonQuery();
            }
            conn.Close();
            Console.WriteLine( "clicy done\n");
           

        }
    }
}
