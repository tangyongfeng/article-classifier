insert into qunNumberOfPeople
select a .QunNum, b.Title ,a. ctn,b .MastQQ, b.createDate ,b. QunText,b .class
 from (
       SELECT QunNum ,count( 1) ctn  FROM dbo. Groups
       group by QunNum )
       a left join dbo.quninfo b
       on a .QunNUm = b .QunNum ;
