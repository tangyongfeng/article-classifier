Apache
下载apache2 解压


sudo apt-get install apache2-dev libapr1-dev libaprutil1-dev libpcre3 libpcre3-dev

configure —prefix=/usr/local/httpd —enable-so  —with-mysql
make
sudo make install

MySQL
sudo apt install mysql-server-5.7

sudo cat /etc/mysql/debian.cnf
里面有默认的用户名和密码

PHP
sudo apt-get install libxml2-dev sqlite3 libsqlite3-dev libbz2-dev libgd-dev libssl-dev  libcurl4-openssl-dev  libonig-dev


./configure --with-apxs2=/usr/local/httpd/bin/apxs --enable-gd --enable-exif --with-bz2 --enable-bcmath  --with-external-gd --with-webp --with-jpeg --with-xpm --with-freetype --with-mysqli --with-zlib  --with-openssl  --with-curl  --enable-mbstring


cp php.ini-development /usr/local/lib/php.ini
